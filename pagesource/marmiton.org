m_overlay = function () {
    var SEL_OVERLAY = 'j_overlay',
        SEL_TITLE = 'j_title',
        SEL_CONTENT = 'j_content',
        SEL_CLOSE = 'j_close',
        SEL_WRAPPER = 'j_wrapper',
        SEL_INTERCEPT = 'j_intercept',
        SEL_ANSWER = 'j_answer',
        CSS_HIDE = 'm_hide',
        CSS_NOSCROLL = 'm_noscroll',
        COOKIE_OVERLAY = 'MARMIMESSAGE',
        answersHandlers = {
            'cancel': function () {
                close();
                return false;
            },
            'no': function () {
                close();
                return false;
            },
            'ok' : function() {
                close();
                return true;
            },
            'yes': function () {
                close();
                return true;
            }
        },
        sizes = {
            'xsmall': 'wrapper_xs',
            'small': 'wrapper_sm',
            'medium': 'wrapper_md',
            'large': 'wrapper_lg'
        },
        answerCallback,
        overlay,
        wrapper,
        title,
        content,
        answers,
        overlayClass,
        center;

    function init() {
        overlay = getElementsByClassName(SEL_OVERLAY, document)[0];
        wrapper = getElementsByClassName(SEL_WRAPPER, overlay)[0];
        title = getElementsByClassName(SEL_TITLE, overlay)[0];
        content = getElementsByClassName(SEL_CONTENT, overlay)[0];
        answers = getElementsByClassName(SEL_ANSWER, overlay);
        
        setEvents();

        if (readCookie(COOKIE_OVERLAY)) {
            show();
            eraseCookie(COOKIE_OVERLAY);
        }
    }

    function setEvents() {
        var close_bttns = getElementsByClassName(SEL_CLOSE, overlay),
            intercept_elms = getElementsByClassName(SEL_INTERCEPT, overlay),
            i = intercept_elms.length,
            answer,
            data;

        for (; i--;) {
            m_bindEvent(intercept_elms[i], 'click', intercept);
        }

        i = close_bttns.length;
        for (; i--;) {
            m_bindEvent(close_bttns[i], 'click', answerClick(close_bttns[i], answersHandlers.cancel));
        }

        i = answers.length;
        for (; i--;) {
            answer = answers[i];
            data = $(answer).data('answer');
            if (!answersHandlers.hasOwnProperty(data)) continue;

            m_bindEvent(answer, 'click', answerClick(answer));
        }
    }

    function answerClick(elm, handler) {
        var self = this,
            data = $(elm).data('answer');
            handler = handler || answersHandlers[data];

        return function () {
            var result = handler.apply(self, arguments);
            if (answerCallback && typeof answerCallback === 'function') {
                answerCallback(result);
            }

            return result;
        };
    }

    function close(evt) {
        removeClass(document.body, CSS_NOSCROLL);
        AddClass(overlay, CSS_HIDE);
        m_preventDefault(evt);
        if(m_events)m_events.publish("m_overlay-closed", {});
        return false;
    }

    function intercept(evt) {
        m_stopPropagation(evt);

        return false;
    }

    function show(opts, callback) {
        var args = opts || {},
            bttns = args.bttns || [],
            answer,
            answerData,
            i;

        //set the wrapper size
        if (args.size && sizes.hasOwnProperty(args.size)) {
            for (var size in sizes) {
                if (!sizes.hasOwnProperty(size)) continue;

                if (args.size === size) {
                    AddClass(wrapper, sizes[size]);
                }
                else {
                    removeClass(wrapper, sizes[size]);
                }
            }
        }

        //show the requested buttons
        i = answers.length;
        for (; i--;) {
            answer = answers[i];
            answerData = $(answer).data('answer');
            if (bttns.indexOf(answerData) === -1) {
                AddClass(answer, CSS_HIDE);
            }
            else {
                removeClass(answer, CSS_HIDE);
            }
        }

        // add overlay class
        if (args.overlayClass) {
            AddClass(overlay, args.overlayClass);
        }

        //add title
        if (args.title) {
            title.innerHTML = args.title;
        }

        //add content
        if (args.content !== undefined) {
            //TODO : gestion d'un contenu html
            content.innerHTML = "";
            if (typeof (args.content) === "object" && args.content.nodeName) {
                content.appendChild(args.content);
            } else {
                content.innerHTML = args.content
            }
        } else {
           // content.innerHTML = "";
        }

        // center
        if (args.center == false)
            removeClass(content, 'm_text_align_center');
        else if (!hasClass(content, 'm_text_align_center'))
            AddClass(content, 'm_text_align_center');


        if (callback && typeof callback === 'function') answerCallback = callback;

        AddClass(document.body, CSS_NOSCROLL);
        removeClass(overlay, CSS_HIDE);
    }

    init();

    return {
        close: close,
        show: show
    };
}();
